def duration_in_mins(datum, city):

    #

def time_of_trip(datum, city):
    """
    
    
    
    
    
    
     datetime 
    
     https://docs.python.org/3/library/datetime.html#strftime-and-strptime-behavior
    """
    
    # 
    if city == 'NYC':
        record_date = datetime.strptime(datum['starttime'], '%m/%d/%Y %H:%M:%S')
    elif city == 'Chicago':
        record_date = datetime.strptime(datum['starttime'], '%m/%d/%Y %H:%M')
    elif city == 'Washington':
        record_date = datetime.strptime(datum['Start date'], '%m/%d/%Y %H:%M')
    else:
        print("Date error!") #other records
        return
    
    week_day_dict = {
        0 : 'Monday',
        1 : 'Tuesday',
        2 : 'Wednesday',
        3 : 'Thursday',
        4 : 'Friday',
        5 : 'Saturday',
        6 : 'Sunday',
    }
    month = record_date.month
    hour = record_date.hour
    day_of_week = week_day_dict[record_date.weekday()]
    
    return (month, hour, day_of_week)

def type_of_user(datum, city):
    """
    
    
    
    
    
    
    """
    
    # 
    user_type = {'Registered': 'Subscriber', 'Casual': 'Customer'}
    if city == 'NYC':
        user_type = datum['usertype']
    elif city == 'Chicago':
        user_type = datum['usertype']
    elif city == 'Washington':
        user_type = user_type[datum['Member Type']]

        return
    
    return user_type

def condense_data(in_file, out_file, city):
    """
    
    
    
    
    
    """
    
    with open(out_file, 'w') as f_out, open(in_file, 'r') as f_in:
        #  csv DictWriter 
        #  "fieldnames" 
        out_colnames = ['duration', 'month', 'hour', 'day_of_week', 'user_type']        
        trip_writer = csv.DictWriter(f_out, fieldnames = out_colnames)
        trip_writer.writeheader()
        
        ##  csv DictReader ##
        trip_reader = csv.DictReader(f_in)

        # 
        for row in trip_reader:
            # 
            new_point = {}

            ## ##
            duration = duration_in_mins(row, city)
            month, hour, day_of_week = time_of_trip(row, city)
            user_type = type_of_user(row, city)
            

            ##  new_point  ##
            ##  DictWriter         ##
            new_point[out_colnames[0]] = duration
            new_point[out_colnames[1]] = month
            new_point[out_colnames[2]] = hour
            new_point[out_colnames[3]] = day_of_week
            new_point[out_colnames[4]] = user_type
            

            ## ##
            ##  https://docs.python.org/3/library/csv.html#writer-objects ##
            trip_writer.writerow(new_point)

def number_of_trips(filename):
    """
    
    
    """
    with open(filename, 'r') as f_in:
        #  csv reader 
        reader = csv.DictReader(f_in)
        
        # 
        n_subscribers = 0
        n_customers = 0
        
        # 
        for row in reader:
            if row['user_type'] == 'Subscriber':
                n_subscribers += 1
            else:
                n_customers += 1
        
        # 
        n_total = n_subscribers + n_customers
        
        # 
        return(n_subscribers, n_customers, n_total)

def get_average_time(filename):
    with open(filename, 'r') as f_in:
        reader = csv.DictReader(f_in)
        
        sum = 0
        count = 0
        for row in reader:
            sum += float(row['duration'])
            count += 1
            
        return sum / count
    
    
def get_precent(filename):
    with open(filename, 'r') as f_in:
        reader = csv.DictReader(f_in)
        
        great_30 = 0
        less_30 = 0
        
        for row in reader:
            if float(row['duration']) <= 30:
                less_30 += 1
            else:
                great_30 += 1
                
        total = less_30 + great_30
        return great_30 / total * 100

def get_user_type(filename):
    with open(filename, 'r') as f_in:
        reader = csv.DictReader(f_in)
        
        sum_subscriber = 0
        count_subscriber = 0
        sum_customer = 0
        count_customer = 0
        for row in reader:
            if row['user_type'] == 'Subscriber':
                sum_subscriber += float(row['duration'])
                count_subscriber += 1
            else:
                sum_customer += float(row['duration'])
                count_customer += 1
            
        average_time_subscriber = sum_subscriber / count_subscriber
        average_time_customer = sum_customer / count_customer
        print("The average time for subscriber: {} min, the average time for customer: {} min"
              .format(average_time_subscriber, average_time_customer))
        
        if average_time_subscriber > average_time_customer:
            return "subscriber"
        if average_time_subscriber < average_time_customer:
            return "customer"
        else:
            return "both subscriber and customer"
        
        
for city, filename in city_info.items():
    print("The longest average riding time in {}: {}".format(city, get_user_type(filename)))
